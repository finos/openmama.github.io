    {
        if (myDateTime == null)

            myDateTime = new MamaDateTime();
            dateTimePointer_i = myDateTime.getPointerVal();
        }        
        _getDateTime ();
        return myDateTime;
    }
    
    public MamaPrice getPrice ()
    {
        if (myPrice == null)

            myPrice = new MamaPrice();

        _getMsg();
